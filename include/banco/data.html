Capturar DIA da semana

SQLSERVER

DATENAME(dw,'2017-09-18') 'DIA_SEMANA'
// retorno Monday

MYSQL

DAYNAME('2017-09-18') 'DIA_SEMANA_EXTENSO'
// retorno Monday
WEEK('2017-09-18') 'DIA_SEMANA_ANO'
	// retorno 0
