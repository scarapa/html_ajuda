Salvando dados de uma tabela

SQLSERVER
DECLARE @SQLStr VARCHAR(MAX);
DECLARE @tabela VARCHAR(200);
SET @tabela = 'TABELA_NOME'
SET @SQLStr='
SELECT 
	''INSERT INTO AVAL_3.'+@tabela+' (cd_producao_animal,ds_nome) VALUES (''+CAST(cd_producao_animal AS VARCHAR(100))+'',''''''+CAST(ds_nome AS VARCHAR(MAX))+'''''');''
FROM AVAL_3.'+@tabela+'';
EXEC(@SQLStr);
