Parse de string para json contendo funcao MAP

CLASSE CLIENTE

class ClienteModel {
  int id;
  String nome;

  ClienteModel({this.id, this.nome, this.listaTelefone});

  Map toMap(){
    var map = new Map<String, dynamic>();
    map["id"] = id;
    map["nome"] = nome;
    return map;
  }

  factory ClienteModel.fromJson(Map<String, dynamic> json) {
    return ClienteModel(
      id: int.parse(json['id'].toString()),
      nome: json['nome'].toString(),
    );
  }
}

CODIGO

import 'dart:convert';
import 'model/cliente_model.dart';

main() {
  String documento = '{"id":1, "nome":"Roberto"} \n{"id":2,"nome":"Tonia"} \n{"id":3,"nome":"Luisa"}';
  // metodo LineSplitter divide por linhas
  LineSplitter.split(documento)
    .map( (linha) {
      ClienteModel cliente = ClienteModel.fromJson( json.decode( linha ) );
      print( cliente );
    }).toList();
}
saida: 
  1 Roberto
  2 Tonia
  3 Luisa
