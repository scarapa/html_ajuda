class Pesssoa 
import 'dart:core';
import 'dart:convert';
class Pessoa {
	int id;
	String nome;

	Pessoa(this.id, this.nome);

	Map toMap(){
		var map = new Map<String, dynamic>();
		map["id"] = id;
		map["nome"] = nome;
		return map;
	}

	@override
	String toString() {
		return "{ id:${this.id}, nome:${this.nome} }";
	}
}
Utilizando o metodo toString() 
void main(List<String> args) {

	List<Pessoa> lista = [new Pessoa(1,"Beto") , new Pessoa(2,"Lu"),new Pessoa(3,"Tonia")];

	List jsonList = lista.map( (listagem) =>  listagem ).toList();
	print(jsonList);
	//[{ id:1, nome:Beto }, { id:2, nome:Lu }, { id:3, nome:Tonia }]
}
Utilizando o metodo toMap() 
void main(List<String> args) {

	List<Pessoa> lista = [new Pessoa(1,"Beto") , new Pessoa(2,"Lu"),new Pessoa(3,"Tonia")];
	
	List jsonList = lista.map( (listagem) => listagem.toMap() ).toList();
	print(jsonList);
	//[{ id:1, nome:Beto }, { id:2, nome:Lu }, { id:3, nome:Tonia }]
}
