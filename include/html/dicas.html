Bloquear submit do formulario com enter

Há 3 maneiras de se fazer isso.

1º onsubmit=return false

<form name="nome_formulario" onsubmit="return false">

2º Criando uma function

//JS
function dasabilitarenter() {
	var tecla = event.keyCode;
	if ((tecla == 13)) {
		return false;
	}
	return tecla;
}

//HTML
<form method='post' action='' onKeyPress='return dasabilitarenter();'>
seus campos vêm aqui....
</form>

3º Criar uma classe

//HTML
<input type='text' name='campo' class='desabilitarEnter'>

//JQUERY
$(document).on('keypress', '.desabilitarEnter', function(event){ 
	if(event.which==13) return false;
});
