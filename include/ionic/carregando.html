import {  LoadingController } from '@ionic/angular';

isLoading = false;

constructor(public loadingCtrl: LoadingController) { }

montarTela(banca='',municipio='',categoria='',ordenacao=''){ 
  	//aqui chama o carregamento
  	this.present();
    
    this.pesquisar().then((retornoLista: any[]) => {
      this.listagem = retornoLista;
    }).catch((e) => {
      console.log(e)
    });
    // aqui acaba de carregar
    this.dismiss();
}

async present() {
    this.isLoading = true;
    return await this.loadingCtrl.create({
      duration: 1000,
    }).then(a => {
      a.present().then(() => {
        if (!this.isLoading) {
          a.dismiss().then(() => console.log('abort presenting'));
        }
      });
    });
}

async dismiss() {
    this.isLoading = false;
    return await this.loadingCtrl.dismiss().then(() => console.log('dismissed'));
}
