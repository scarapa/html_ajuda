Data pra inteiro

function converterDataParaInteiro( data ){
    data = data.toString().split("/");
    var data1 = new Date(data[2], data[1]-1, data[0]);    
    return data1.getTime();
}

Função para comparar datas

function compararDatas(data1 , data2){
    data1 = converterDataParaInteiro(data1);
    data2 = converterDataParaInteiro(data2);
    if(data1 > data2) {
        return 1;
    }else if( data1 == data2)
        return 2;
    }else{
        return 3;
    }
}

Função converta data padrão BRASILEIRO para padrão NORTE AMERICANO(banco)


function converterDataBrToEua(data){
    var data2 = data.split("/");
    if(data2.length ==3){
        return data2[2]+'-'+data2[1]+'-'+data2[0] ;
    }
    return ;
}

Função retorno mes(NUMERICO) da data


function buscarMesData( data ){
    data = converterDataBrToEua(data);
    var d = new Date(data);
    var mes = d.getMonth()+1;
    return mes;
}

Função retorno dia da semana(NUMERICO)


function buscarDiaDaSemanaInteiro(data){
  var data = converterDataBrToEua(data);
  var d = new Date(data);
  return parseInt( d.getDay() ) ;
}

Função retorno dia da semana, descrição longa


function buscarDiaDaSemanaDescricaoLonga(data){
  var semana = ["Domingo","Segunda","Terça","Quarta","Quinta","Sexta","Sabado"];		
  var data = converterDataBrToEua(data);
  return semana[ buscarDiaDaSemanaInteiro(data) ];  
}

Função retorno dia da semana, descrição curta


function buscarDiaDaSemanaDescricaoCurta(data){
  var semana = ["dom","seg","ter","qua","qui","sex","sab"];		
  var data = converterDataBrToEua(data);
  return semana[ buscarDiaDaSemanaInteiro(data) ];  
}
