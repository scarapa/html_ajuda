Objetivo Ã© armazenar o texto do option selecionado no select 
HTML
<select id="id_marca" onkeypress="setar(this)" onchange="setar(this)">
	<option>Fiat</option>
	<option>Ford</option>	
	<option>Peugeot</option>

</select>
<input type="text" id="ds_marca">

JS
<script type="text/javascript">
	function setar(objeto){
		document.getElementById("ds_marca").value = objeto.options[objeto.selectedIndex].text ;
	}
</script>
