LOAD

Funcionalidade semelhante ao LOAD do jQuery 
JS
window.onload = function(e){
	document.getElementById('wrapper').innerHTML =  loadPage('contato.html')

	function loadPage(href){
    	var xmlhttp = new XMLHttpRequest();
    	xmlhttp.open("GET", href, false);
    	xmlhttp.send();
		return xmlhttp.responseText;
	}

    //MENU
    var elementos = document.querySelectorAll(".sidebar li");
    for (var i = 0; i < elementos.length; i++) {

        elementos[i].addEventListener('click', function (event) {
            link = this.innerText.toLowerCase();
            link = link+".html";
            document.getElementById("wrapper").innerHTML = loadPage(link);
        });
    } 	
}
