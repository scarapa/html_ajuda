VARIAVEIS NA URL

EXEMPLO: http://idiota.com.br/?modulo=informatica&conteudo=jquery&capitulo=3 

var getUrlParameter = function getUrlParameter(sParam) {
    var sPageURL = decodeURIComponent(window.location.search.substring(1)),
        sURLVariables = sPageURL.split('&'),
        sParameterName,
        i;

    for (i = 0; i < sURLVariables.length; i++) {
        sParameterName = sURLVariables[i].split('=');

        if (sParameterName[0] === sParam) {
            return sParameterName[1] === undefined ? true : sParameterName[1];
        }
    }
};
Capturando as variaveis: 
var modulo = getUrlParameter('modulo');
var conteudo = getUrlParameter('conteudo');
var capitulo = getUrlParameter('capitulo');
