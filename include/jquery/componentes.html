CHECKBOX

VERIFICAR MARCACAO

$("#idComponente").click( function(){
    //SE ESTIVER CHECHADO RETORNA TRUE
    if ( $(this).is(':checked') ){
        //MARCA TODOS OS INPUTS
        $("input[type='checkbox']").prop( "checked", true );
    }else{
        //DESMARCA TODOS OS INPUTS        
        $("input[type='checkbox']").prop( "checked", false );
    }
});
