#!/bin/bash
# desmontando possiveis particoes
sudo umount /dev/sdb1

# montando particao
sudo ntfs-3g /dev/sdb1 /media/drive

# iniciando servicos lampp
sudo /opt/lampp/lampp start

# arquivo de log
DATA=$(date +%Y%m%d-%H%M)
touch /home/beto/log/${DATA}.txt
Setando o arquivo como executavel 
$ chmod +x /etc/init.d/servidor.sh
carregando o arquivo 
$ source /etc/init.d/servidor.sh
