function divisao($n1,$n2) {
    if ( (empty($n2)) || ($n2==0)) {
        throw new Exception('Divisão por zero.');
    }
    return $n1/$n2;
}

try {
    echo divisao(5,0)."<br>";
} catch (Exception $e) {
    echo 'Exceção capturada: ',  $e->getMessage(), "\n";
}
